kill_running_process() {
  arg1=$1
  arg2=$2
  arg3=$3
  # shellcheck disable=SC2046
  if [ "$arg1" -a "$arg2" -a "$arg3" ]; then
    kill -9 "$arg3"
  elif [[ ! -z "$arg1" ]]; then
    kill -9 "$arg1"
  else
    echo "Nothing to kill"
  fi

}

check_process_running() {
  arg1=$1
  arg2=$2
  arg3=$3
  # shellcheck disable=SC2107
  # shellcheck disable=SC2202
  # shellcheck disable=SC1026
  if [ "$arg1" -a "$arg2" -a "$arg3" ]; then
    returnVal=0
  elif [[ ! -z "$arg1" ]]; then
    returnVal=0
  else
    returnVal=1
  fi
  return $returnVal
}

start_service() {
  SERVICE_NAME=$1
  PATH_TO_JAR=$2
  PATH_TO_PROP=$3

  PID=$(pgrep -f -d ' ' $SERVICE_NAME-0.0.1-SNAPSHOT.jar)
  #  PID=$(pgrep -f $SERVICE_NAME)

  check_process_running $PID
  running_status=$?

  if [ "$running_status" == 0 ]; then
    echo "$SERVICE_NAME is already running ..."
  else
    echo "Starting $SERVICE_NAME ..."
    if [ ! -f "$PATH_TO_JAR" ]; then
      echo "$SERVICE_NAME is not a valid jar..."
    else
      nohup java -jar "$PATH_TO_JAR" --spring.config.location="$PATH_TO_PROP" &
      echo "$SERVICE_NAME started ..."
    fi
  fi
}
stop_service() {
  SERVICE_NAME=$1

  PID=$(pgrep -f -d ' ' $SERVICE_NAME-0.0.1-SNAPSHOT.jar)
  #  PID=$(pgrep -f $SERVICE_NAME)

  check_process_running $PID
  running_status=$?

  if [ "$running_status" == 0 ]; then
    echo "$SERVICE_NAME is stopping ..."
    kill_running_process $PID
    killing_result=$?
    if [ "$killing_result" == 0 ]; then
      echo "$SERVICE_NAME stopped ..."
    else
      echo "$SERVICE_NAME refused to stop ..."
    fi
  else
    echo "$SERVICE_NAME is not running ..."
  fi
  return
}

restart_service() {
  SERVICE_NAME=$1
  PATH_TO_JAR=$2
  PATH_TO_PROP=$3

  PID=$(pgrep -f -d ' ' $SERVICE_NAME-0.0.1-SNAPSHOT.jar)
  #  PID=$(pgrep -f $SERVICE_NAME)

  check_process_running $PID
  running_status=$?

  if [ "$running_status" == 0 ]; then
    echo "$SERVICE_NAME stopping ..."
    kill_running_process $PID
    killing_result=$?
    if [ "$killing_result" == 0 ]; then

      echo "$SERVICE_NAME stopped ..."
      echo "$SERVICE_NAME starting ..."
      nohup java -jar "$PATH_TO_JAR" --spring.config.location="$PATH_TO_PROP" &
      echo "$SERVICE_NAME started ..."
    else
      echo "$SERVICE_NAME refused to stop ..."
    fi
  else
    echo "$SERVICE_NAME is not running ..."
  fi
}

check_service_status() {

  SERVICE_NAME=$1

  PID=$(pgrep -f -d ' ' $SERVICE_NAME-0.0.1-SNAPSHOT.jar)
  #  PID=$(pgrep -f $SERVICE_NAME)
  check_process_running $PID
  running_status=$?

  if [ $running_status == 0 ]; then
    MEM=$(pmap $PID | grep total)
    echo "[Running] : $SERVICE_NAME service ... [$PID][$MEM]"
  else
    echo "[Stopped] : $SERVICE_NAME service ..." $PID
  fi
  echo ""
}
